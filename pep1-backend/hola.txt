pipeline {
    agent any
    tools {
        maven "Maven"
    }
    stages {
        stage("Build JAR file") {
            steps {
                checkout scmGit(branches: [[name: '*/main']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/Just-NN/Tingeso1']])
                sh "mvn clean install"
            }
        }

    }
    post {
        always {
            sh "docker logout"
        }
    }
}
